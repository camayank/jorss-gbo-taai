['Additional Medicare Tax (0.9%)', fmt(addMcTax), false] : null, ['Total Self-Employment Tax', fmt(totalSETax), true], ['Deductible Half (reduces your AGI)', fmt(deductibleHalf), false], ['Potential QBI Deduction (20%)', fmt(qbiDeduction), false], ['Estimated Quarterly Payment', fmt(quarterly), false], ].filter(Boolean); document.getElementById('result-rows').innerHTML = rows.map(([label, value, highlight]) => `
${label} ${value}
