        self,
        session_id: str,
        include_scenarios: bool = True,
        include_multi_year: bool = True,
        include_state: bool = True

        """
        Generate complete advisory computation report.

        Returns structured data for all 7 sections:
        1. Executive Summary Dashboard
        2. Form 1040 Line-by-Line Computation
        3. Scenario Comparison Analysis
        4. Tax Bracket Optimization
        5. Multi-Year Tax Projection
        6. Strategic Recommendations
        7. Supporting Schedules & Worksheets
        """

        # Get session data
        session_data = await self.get_session_data(session_id)

        # Section 1: Executive Summary
        executive_summary = await self.generate_executive_summary(session_data)

        # Section 2: Form 1040 Line-by-Line

        bracket_optimization = await self.generate_bracket_optimization(session_data)

        # Section 5: Multi-Year Projection (if requested)

    session_id: str,
    include_scenarios: bool = True,
    include_multi_year: bool = True,
    include_state: bool = True,
    format: str = "json",  # json, pdf, html
    background_tasks: BackgroundTasks = None
):

    Args:
        session_id: Tax session identifier

        include_multi_year: Include 3-5 year projections
        include_state: Include state tax analysis
        format: Output format (json, pdf, html)

    Returns:
        Complete advisory report with 7 sections:
        1. Executive Summary Dashboard
        2. Form 1040 Line-by-Line Computation
        3. Scenario Comparison Analysis
        4. Tax Bracket Optimization
        5. Multi-Year Tax Projection

            return report

        raise HTTPException(
            status_code=500,
            detail=f"Failed to generate advisory report: {str(e)}"
